def rank_hubs_by_distance(self):
    """BFS from end_hub to compute distance rank for each hub."""
    q = deque([(self.end_hub, 0)])
    ranks = {self.end_hub: 0}
    while q:
        hub, dist = q.popleft()
        for neigh in hub.neighbour_hubs:
            if neigh not in ranks:
                ranks[neigh] = dist + 1
                q.append((neigh, dist + 1))
    for hub in self.hubs:
        hub.rank = ranks.get(hub, float('inf'))
